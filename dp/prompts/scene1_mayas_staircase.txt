# Climbing Stairs — Scene 1 trace (panels 1-7)
def climbStairs(n):
    if n <= 1: return 1                    # panel 2: base case
    dp = [0] * (n + 1)
    dp[0] = 1                              # panel 2: golden chalk
    dp[1] = 1                              # panel 2: golden chalk
    for i in range(2, n + 1):              # panels 4-7: the climb
        dp[i] = dp[i-1] + dp[i-2]         # panel 3-7: look back, add, write
    return dp[n]                           # panel 7: summit

# Space-optimized version (panels 8-9)
def climbStairs_optimized(n):
    if n <= 1: return 1
    prev2, prev1 = 1, 1                   # panel 8: two chalk pieces
    for i in range(2, n + 1):
        curr = prev1 + prev2              # panel 9: compute
        prev2 = prev1                     # panel 9: shift
        prev1 = curr                      # panel 9: shift
    return prev1
