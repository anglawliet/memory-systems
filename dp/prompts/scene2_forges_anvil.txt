**Footer — "The Target-Building DP Pattern:"**
1. **"Golden Zero"**: Base case. dp[0] = 0 (or 1 for counting). The foundation plate.
2. **"Mold by Mold"**: Outer loop over coins/items. Pick one mold, process the entire shelf with it.
3. **"Hammer Every Slot"**: Inner loop from coin to amount. For each slot, check: is dp[a - coin] + 1 better than what's here? If yes, SLAM — swap the plate.

# Coin Change — Scene 2 trace (panels 1-8)
def coinChange(coins, amount):
    dp = [float('inf')] * (amount + 1)    # panel 1: rusty shelf
    dp[0] = 0                              # panel 1: golden zero

    for coin in coins:                     # panels 2,4,7: pick a mold
        for a in range(coin, amount + 1):  # panels 3,5-6,8: hammer each slot
            dp[a] = min(dp[a], dp[a - coin] + 1)  # panels 3,5,8: compare and swap

    if dp[amount] == float('inf'):         # panel 9: impossible amount
        return -1
    return dp[amount]                      # panel 10: read the answer plate

# Coin Change II — count combinations (panel 10, problem 2)
def change(amount, coins):
    dp = [0] * (amount + 1)
    dp[0] = 1                              # one way to make zero: use nothing
    for coin in coins:
        for a in range(coin, amount + 1):
            dp[a] += dp[a - coin]          # count, don't minimize
    return dp[amount]

# Target Sum — subset sum variant (panel 10, problem 3)
def findTargetSumWays(nums, target):
    total = sum(nums)
    if (total + target) % 2 != 0: return 0
    subset = (total + target) // 2
    dp = [0] * (subset + 1)
    dp[0] = 1
    for num in nums:
        for s in range(subset, num - 1, -1):  # reverse to avoid reuse
            dp[s] += dp[s - num]
    return dp[subset]
