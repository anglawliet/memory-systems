# Count Palindromic Substrings — Scene 3 trace (panels 1-11)
def countSubstrings(s):                        # panel 1: the mirror corridor
    count = 0
    for center in range(len(s)):               # panels 2, 5: stand on each center
        # Odd-length palindromes (single center)
        l, r = center, center                  # panel 2: one-tile stance
        while l >= 0 and r < len(s) and s[l] == s[r]:  # panels 6-7: green glow
            count += 1                         # panels 2, 6, 7: click counter
            l -= 1                             # panel 6: left arm reaches further
            r += 1                             # panel 6: right arm reaches further
        # l < 0 or r >= len(s)                 # panels 3, 8: off the edge, STOP

        # Even-length palindromes (double center)
        l, r = center, center + 1             # panels 4, 9: between-tiles stance
        while l >= 0 and r < len(s) and s[l] == s[r]:  # panel 4: green glow
            count += 1                         # panels 4, 9: click counter
            l -= 1
            r += 1
        # s[l] != s[r]                        # panel 10: RED FLASH, mismatch

    return count                               # panel 11: final count = 9

# Longest Palindromic Substring — same expansion, track longest (panel 12, problem 1)
def longestPalindrome(s):
    result = ""
    for center in range(len(s)):
        # Odd-length
        l, r = center, center
        while l >= 0 and r < len(s) and s[l] == s[r]:
            if r - l + 1 > len(result):        # panel 7: longest = "aabaa"
                result = s[l:r+1]
            l -= 1
            r += 1
        # Even-length
        l, r = center, center + 1
        while l >= 0 and r < len(s) and s[l] == s[r]:
            if r - l + 1 > len(result):
                result = s[l:r+1]
            l -= 1
            r += 1
    return result
