# Longest Common Subsequence — Scene 4 trace (panels 1-8)
def longestCommonSubsequence(text1, text2):
    m, n = len(text1), len(text2)
    dp = [[0] * (n + 1) for _ in range(m + 1)]   # panel 1: empty grid, white zeros

    for i in range(1, m + 1):                     # panels 2-7: row by row
        for j in range(1, n + 1):                 # panels 2-7: column by column
            if text1[i-1] == text2[j-1]:          # panels 2,4,7: MATCH
                dp[i][j] = dp[i-1][j-1] + 1      # panels 2,4,7: golden diagonal + 1
            else:                                  # panels 3,5,6: NO MATCH
                dp[i][j] = max(dp[i-1][j], dp[i][j-1])  # panels 3,5,6: silver from max(up, left)

    return dp[m][n]                               # panel 7: bottom-right glows

# Unique Paths — same grid, different rule (panel 9, problem 1)
def uniquePaths(m, n):
    dp = [[1] * n for _ in range(m)]              # first row and col all 1s
    for i in range(1, m):
        for j in range(1, n):
            dp[i][j] = dp[i-1][j] + dp[i][j-1]   # sum up + left (no diagonal)
    return dp[m-1][n-1]

# Edit Distance — three-way comparison (panel 9, problem 3)
def minDistance(word1, word2):
    m, n = len(word1), len(word2)
    dp = [[0] * (n + 1) for _ in range(m + 1)]
    for i in range(m + 1): dp[i][0] = i           # base: delete all
    for j in range(n + 1): dp[0][j] = j           # base: insert all
    for i in range(1, m + 1):
        for j in range(1, n + 1):
            if word1[i-1] == word2[j-1]:
                dp[i][j] = dp[i-1][j-1]           # match: diagonal, no cost
            else:
                dp[i][j] = 1 + min(dp[i-1][j-1],  # replace (diagonal)
                                   dp[i-1][j],     # delete (up)
                                   dp[i][j-1])     # insert (left)
    return dp[m][n]

# Distinct Subsequences (panel 10, problem 5)
def numDistinct(s, t):
    m, n = len(s), len(t)
    dp = [[0] * (n + 1) for _ in range(m + 1)]
    for i in range(m + 1): dp[i][0] = 1           # empty t: one way (delete all)
    for i in range(1, m + 1):
        for j in range(1, n + 1):
            dp[i][j] = dp[i-1][j]                 # skip s[i-1]
            if s[i-1] == t[j-1]:
                dp[i][j] += dp[i-1][j-1]          # match: also count diagonal
    return dp[m][n]
