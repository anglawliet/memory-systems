**Footer — "The State Machine DP Pattern:"**
1. **"Three Cards"**: Define your states. Each state card tracks the best value achievable while IN that state. Initialize Day 0.
2. **"Transition Arrows"**: Each day, compute new state values from PREVIOUS day's states using the transition rules. New values depend on old values — never update in place.
3. **"Max of Finals"**: The answer is the best among all VALID ending states (sold or rest — you wouldn't end while still holding).

# Best Time to Buy & Sell Stock with Cooldown — Scene 5 trace (panels 1-7)
def maxProfit(prices):
    if not prices: return 0

    # Day 0: initialize three states              # panels 1-2: cards dealt
    hold = -prices[0]   # bought on day 0          # panel 2: red card, chip below table
    sold = 0            # haven't sold yet          # panel 2: green card, flat
    rest = 0            # haven't done anything      # panel 2: blue card, flat

    for price in prices[1:]:                        # panels 3-6: day by day
        new_hold = max(hold, rest - price)          # panel 3: max(keep holding, buy from rest)
                                                    # panel 5: rest-price=1-0=1, BUY THE DIP!
        new_sold = hold + price                     # panel 4: hold+price=sell today
                                                    # panel 6: 1+2=3, the winning play!
        new_rest = max(rest, sold)                  # panel 4: cooldown absorbs sold→rest
                                                    # panel 8: MUST pass through rest!
        hold, sold, rest = new_hold, new_sold, new_rest  # swap all at once

    return max(sold, rest)                          # panel 7: max(3, 2) = 3

# Longest Increasing Subsequence — O(n^2) DP (panel 9, problem 2)
def lengthOfLIS(nums):
    if not nums: return 0
    dp = [1] * len(nums)                            # every element is a subsequence of length 1
    for i in range(1, len(nums)):
        for j in range(i):
            if nums[j] < nums[i]:                   # can extend subsequence ending at j
                dp[i] = max(dp[i], dp[j] + 1)       # take the best extension
    return max(dp)

# Longest Increasing Subsequence — O(n log n) patience sorting (panel 9, problem 2)
import bisect
def lengthOfLIS_fast(nums):
    tails = []                                       # smallest tail of each pile length
    for num in nums:
        pos = bisect.bisect_left(tails, num)
        if pos == len(tails):
            tails.append(num)                        # new longest subsequence
        else:
            tails[pos] = num                         # replace to keep tails small
    return len(tails)

# Burst Balloons — interval DP (panel 9, problem 3)
def maxCoins(nums):
    nums = [1] + nums + [1]                          # pad with 1s on both sides
    n = len(nums)
    dp = [[0] * n for _ in range(n)]

    for length in range(2, n):                       # interval length
        for l in range(0, n - length):               # left boundary
            r = l + length                           # right boundary
            for k in range(l + 1, r):                # last balloon to burst in (l, r)
                dp[l][r] = max(dp[l][r],
                    dp[l][k] + dp[k][r] + nums[l] * nums[k] * nums[r])
    return dp[0][n - 1]
