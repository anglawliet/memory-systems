parent = {'Kyle': 'Kyle', ...}   # panel 1
rank = {'Kyle': 0, ...}          # panel 1
def find(x):
    if parent[x] != x:
        parent[x] = find(parent[x])  # Step 2: Path compression (panel 9)
    return parent[x]
def union(a, b):
    root_a = find(a)
    root_b = find(b)
    if root_a == root_b: return       # already same company
    if rank[root_a] < rank[root_b]:   # different ranks (panel 6)
        parent[root_a] = root_b
    elif rank[root_a] > rank[root_b]: # different ranks (panel 6)
        parent[root_b] = root_a
    else:
        parent[root_b] = root_a       # exact piece of code for rank bump (panels 2, 3, 4)
        rank[root_a] += 1
